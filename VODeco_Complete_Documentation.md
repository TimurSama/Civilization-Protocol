```python
def calculate_irr(cash_flows: List[float]) -> float:
    """
    –†–∞—Å—á–µ—Ç –≤–Ω—É—Ç—Ä–µ–Ω–Ω–µ–π –Ω–æ—Ä–º—ã –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏

    """
    def npv(rate):
        return sum(cf / (1 + rate) ** i for i, cf in enumerate(cash_flows))
    
    # –ò—Å–ø–æ–ª—å–∑—É–µ–º –º–µ—Ç–æ–¥ –ù—å—é—Ç–æ–Ω–∞ –¥–ª—è –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è IRR
    rate = 0.1
    for _ in range(100):
        rate = rate - npv(rate) / npv(rate)
        if abs(npv(rate)) < 0.0001:
            return rate
    return rate
```
